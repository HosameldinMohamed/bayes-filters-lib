113 Eigen::VectorXd GPFCorrection::sampleFromProposal(const Eigen::VectorXd& mean, const Eigen::MatrixXd& covariance)
114 {
115  /* Evaluate the square root of the state covariance matrix using the LDL' decomposition
116  (it can be used even if the covariance matrix is positive semidefinite). */
117  LDLT<MatrixXd> chol_ldlt(covariance);
118  MatrixXd sqrt_P = (chol_ldlt.transpositionsP() * MatrixXd::Identity(mean.size(), mean.size())).transpose() *
119  chol_ldlt.matrixL() *
120  chol_ldlt.vectorD().real().cwiseSqrt().asDiagonal();
121 
122  /* Sample i.i.d standard normal univariates. */
123  VectorXd rand_vectors(mean.size());
124  for (int i = 0; i < rand_vectors.size(); i++)
125  rand_vectors(i) = gaussian_random_sample_();
126 
127  /* Return a sample from a normal multivariate having mean `mean` and covariance 'covariance'. */
128  return mean + sqrt_P * rand_vectors;
129 }
