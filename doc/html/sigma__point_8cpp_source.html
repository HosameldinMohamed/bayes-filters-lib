28  const std::size_t n,
29  const double alpha,
30  const double beta,
31  const double kappa,
32  Ref<VectorXd> weight_mean,
33  Ref<VectorXd> weight_covariance,
34  double& c
35 )
36 {
37  double lambda = std::pow(alpha, 2.0) * (n + kappa) - n;
38 
39  for (int j = 0; j < ((2 * n) + 1); ++j)
40  {
41  if (j == 0)
42  {
43  weight_mean(j) = lambda / (n + lambda);

49  weight_covariance(j) = weight_mean(j);

57 MatrixXd bfl::sigma_point::sigma_point(const GaussianMixture& state, const double c)
58 {
59  MatrixXd sigma_points(state.dim, ((state.dim * 2) + 1) * state.components);
60 
61  for (std::size_t i = 0; i < state.components; i++)
62  {
63  JacobiSVD<MatrixXd> svd = state.covariance(i).jacobiSvd(ComputeThinU);
64 
65  MatrixXd A = svd.matrixU() * svd.singularValues().cwiseSqrt().asDiagonal();
66 
67  Ref<MatrixXd> sp = sigma_points.middleCols(((state.dim * 2) + 1) * i, ((state.dim * 2) + 1));
68 
69  sp << VectorXd::Zero(state.dim), std::sqrt(c) * A, -std::sqrt(c) * A;
70 
71  if (state.dim_linear > 0)
72  sp.topRows(state.dim_linear).colwise() += state.mean(i).topRows(state.dim_linear);
73 
74  if (state.dim_circular > 0)
75  sp.middleRows(state.dim_linear, state.dim_circular) = directional_add(sp.middleRows(state.dim_linear, state.dim_circular), state.mean(i).middleRows(state.dim_linear, state.dim_circular));
76 
77  if (state.dim_noise > 0)
78  sp.bottomRows(state.dim_noise).colwise() += state.mean(i).bottomRows(state.dim_noise);
79  }
80 
81  return sigma_points;
82 }

85 std::tuple<bool, GaussianMixture, MatrixXd> bfl::sigma_point::unscented_transform
86 (
87  const GaussianMixture& input,
88  const UTWeight& weight,
89  FunctionEvaluation function
90 )
91 {
92  /* Sample sigma points. */
93  MatrixXd input_sigma_points = sigma_point::sigma_point(input, weight.c);
94 
95  /* Propagate sigma points */
96  Data fun_data;
97  bool valid_fun_data;
98  bfl::sigma_point::OutputSize output_size;
99  std::tie(valid_fun_data, fun_data, output_size) = function(input_sigma_points);
100 
101  /* Stop here if function evaluation failed. */
102  if (!valid_fun_data)
103  return std::make_tuple(false, GaussianMixture(), MatrixXd(0, 0));
104 
105  /* For now casting Data to MatrixXd. */
106  MatrixXd prop_sigma_points = bfl::any::any_cast<MatrixXd&&>(std::move(fun_data));
107 
108  /* Initialize transformed gaussian. */
109  GaussianMixture output(input.components, prop_sigma_points.rows());
110 
111  /* Initialize cross covariance matrix. */
112  MatrixXd cross_covariance(input.dim, output.dim * output.components);
113 
114  /* Process all the components of the mixture. */
115  std::size_t base = ((input.dim * 2) + 1);
116  for (std::size_t i = 0; i < input.components; i++)
117  {
118  Ref<MatrixXd> input_sigma_points_i = input_sigma_points.middleCols(base * i, base);
119  Ref<MatrixXd> prop_sigma_points_i = prop_sigma_points.middleCols(base * i, base);
120 
121  /* Evaluate the mean. */
122  output.mean(i).topRows(output_size.first).noalias() = prop_sigma_points_i.topRows(output_size.first) * weight.mean;
123  output.mean(i).bottomRows(output_size.second) = directional_mean(prop_sigma_points_i.bottomRows(output_size.second), weight.mean);
124 
125  /* Evaluate the covariance. */
126  prop_sigma_points_i.topRows(output_size.first).colwise() -= output.mean(i).topRows(output_size.first);
127  prop_sigma_points_i.bottomRows(output_size.second) = directional_sub(prop_sigma_points_i.bottomRows(output_size.second), output.mean(i).bottomRows(output_size.second));
128  output.covariance(i).noalias() = prop_sigma_points_i * weight.covariance.asDiagonal() * prop_sigma_points_i.transpose();
129 
130  /* Evaluate the input-output cross covariance matrix
131  (noise components in the input are not considered). */
132  Ref<MatrixXd> cross_covariance_i = cross_covariance.middleCols(output.dim * i, output.dim);
133  input_sigma_points_i.topRows(input.dim_linear).colwise() -= input.mean(i).topRows(input.dim_linear);
134  input_sigma_points_i.middleRows(input.dim_linear, input.dim_circular) = directional_sub(input_sigma_points_i.middleRows(input.dim_linear, input.dim_circular), input.mean(i).middleRows(input.dim_linear, input.dim_circular));
135  cross_covariance_i.noalias() = input_sigma_points_i.topRows(input.dim_linear + input.dim_circular) * weight.covariance.asDiagonal() * prop_sigma_points_i.transpose();
136  }
137 
138  return std::make_tuple(true, output, cross_covariance);
139 }
