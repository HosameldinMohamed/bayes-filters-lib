54 void ResamplingWithPrior::resample(const ParticleSet& cor_particles, ParticleSet& res_particles, Ref<VectorXi> res_parents)
55 {
56  int num_prior_particles = static_cast<int>(std::floor(cor_particles.state().cols() * prior_ratio_));
57  int num_resample_particles = cor_particles.state().cols() - num_prior_particles;
58 
59  /* Consider two subsets of particles. */
60  ParticleSet res_particles_left(num_prior_particles, cor_particles.dim_linear, cor_particles.dim_circular);
61  ParticleSet res_particles_right(num_resample_particles, cor_particles.dim_linear, cor_particles.dim_circular);
62  Ref<VectorXi> res_parents_right(res_parents.tail(num_resample_particles));
63 
64  /* Copy particles to be resampled in a temporary. */
65  ParticleSet tmp_particles(num_resample_particles, cor_particles.dim_linear, cor_particles.dim_circular);
66  int j = 0;
67  for (std::size_t i : sort_indices(cor_particles.weight().array().exp()))
68  {
69  if (j >= num_prior_particles)
70  {
71  tmp_particles.state(j - num_prior_particles) = cor_particles.state(i);
72  tmp_particles.mean(j - num_prior_particles) = cor_particles.mean(i);
73  tmp_particles.covariance(j - num_prior_particles) = cor_particles.covariance(i);
74  tmp_particles.weight(j - num_prior_particles) = cor_particles.weight(i);
75  }
76  j++;
77  }
78 
79  /* Normalize weights using LogSumExp. */
80  tmp_particles.weight().array() -= utils::log_sum_exp(tmp_particles.weight());
81 
82  /* Resample from tmp_particles. */
83  Resampling::resample(tmp_particles, res_particles_right, res_parents_right);
84  res_parents_right.array() += num_prior_particles;
85 
86  /* Initialize from scratch num_prior_particles particles. */
87  init_model_->initialize(res_particles_left);
88 
89  /* Merge the two subset together. */
90  res_particles = std::move(res_particles_left + res_particles_right);
91 
92  /* Reset weights.*/
93  res_particles.weight().setConstant(-log(cor_particles.state().cols()));
94 
95  /* Since num_prior_particles were created from scratch,
96  they do not have a parent. */
97  res_parents.head(num_prior_particles).array() = -1;
98 }
