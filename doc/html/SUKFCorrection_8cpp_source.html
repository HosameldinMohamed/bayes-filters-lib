104

108  return;
109  }

113 
114  /* From now on using equations from the paper:
115  Barfoot, T., McManus, C. (2011),
116  'A Serial Approach to Handling High-Dimensional Measurements in the Sigma-Point Kalman Filter.',
117  Science and Systems VII,
118  MIT Press */
119 
120  /* Process all the components in the mixture. */
121  MatrixXd sqrt_ut_weight = ut_weight_.covariance.array().sqrt().matrix().asDiagonal();
122  for (size_t i = 0; i < pred_state.components; i++)
123  {
124  /* Compose square root of the measurement covariance matrix.
125  IV.C.2.b */
126  Ref<MatrixXd> Y = prop_sigma_points.middleCols(size_sigmas * i, size_sigmas);
127 
128  /* Shift w.r.t. the mean. */
129  Y.colwise() -= pred_mean.col(i);
130 
131  /* Weight using square root of unscented transform weight. */
132  Y *= sqrt_ut_weight;
133 
134  /* Compose matrix C and vector d by cycling over all the sub-vector of the measurement
135  IV.C.2.c */
136  MatrixXd C_inv = MatrixXd::Identity(size_sigmas, size_sigmas);
137  VectorXd d = VectorXd::Zero(size_sigmas);
138 
139  for (size_t j = 0; j < (meas_size) / measurement_sub_size_; j++)
140  {
141  MatrixXd tmp(size_sigmas, measurement_sub_size_);
142  tmp.noalias() = Y.middleRows(measurement_sub_size_ *j, measurement_sub_size_).transpose() * getNoiseCovarianceMatrix(j).inverse();
143 
144  C_inv += tmp * Y.middleRows(measurement_sub_size_ *j, measurement_sub_size_);
145 
146  d += tmp * innovations.col(i).middleRows(measurement_sub_size_ *j, measurement_sub_size_);
147  }
148 
149  /* Process input sigma points.
150  IV.C.3 */
151  Ref<MatrixXd> X = input_sigma_points.middleCols(size_sigmas * i, size_sigmas);
152  X.topRows(pred_state.dim_linear).colwise() -= pred_state.mean(i).topRows(pred_state.dim_linear);
153  X.bottomRows(pred_state.dim_circular) = directional_sub(X.bottomRows(pred_state.dim_circular), pred_state.mean(i).bottomRows(pred_state.dim_circular));
154  X *= sqrt_ut_weight;
155 
156  C_inv = C_inv.inverse();
157 
158  /* Evaluate the filtered mean.
159  IV.C.4 */
160  corr_state.mean(i) = pred_state.mean(i) + X * C_inv * d;
161 
162  /* Evaluate the filtered covariance.
163  IV.C.4 */
164  corr_state.covariance(i) = X * C_inv * X.transpose();
165  }
