136 bool GaussianMixture::augmentWithNoise(const Eigen::Ref<const Eigen::MatrixXd>& noise_covariance_matrix)
137 {
138  /* Augment each state with a noise component having zero mean
139  and given covariance matrix. */
140 
141  /* Check that covariance matrix is square. */
142  if (noise_covariance_matrix.rows() != noise_covariance_matrix.cols())
143  return false;
144 
145  dim_noise = noise_covariance_matrix.rows();
146  dim += dim_noise;
147 
148  /* Add zero mean noise to each mean. */
149  mean_.conservativeResize(dim, NoChange);
150  mean_.bottomRows(dim_noise) = MatrixXd::Zero(dim_noise, components);
151 
152  /* Resize covariance matrix. */
153  covariance_.conservativeResizeLike(MatrixXd::Zero(dim, dim * components));
154 
155  /* Move old covariance matrices from right to left to avoid aliasing.
156 
157  Note that the covariance matrix of the 0-th coomponent,
158  i.e. in the top-left corner of the matrix covariance_,
159  is already in the correct place.
160  */
161  std::size_t dim_old = dim_linear + dim_circular;
162  for (std::size_t i = 0; i < (components - 1); i++)
163  {
164  std::size_t i_index = components - 1 - i;
165 
166  Ref<MatrixXd> new_block = covariance_.block(0, i_index * dim, dim_old, dim_old);
167  Ref<MatrixXd> old_block = covariance_.block(0, i_index * dim_old, dim_old, dim_old);
168 
169  /* Swap columns from to right to left to avoid aliasing. */
170  for (std::size_t j = 0; j < dim_old; j++)
171  {
172  std::size_t j_index = dim_old - 1 - j;
173 
174  new_block.col(j_index).swap(old_block.col(j_index));
175  }
176  }
177 
178  for (std::size_t i = 0; i < components; i++)
179  {
180  /* Copy the noise covariance matrix in the bottom-right block of
181  each covariance matrix. */
182  covariance_.block(dim_old, i * dim + dim_old, dim_noise, dim_noise) = noise_covariance_matrix;
183 
184  /* Clean part of the matrix that should be zero. */
185  covariance_.block(0, i * dim + dim_old, dim_old, dim_noise) = MatrixXd::Zero(dim_old, dim_noise);
186  }
187 
188  return true;
189 }
