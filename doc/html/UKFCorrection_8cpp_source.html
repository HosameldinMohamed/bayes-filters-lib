110  MatrixXd y_p = std::move(pred_meas.mean());

121

123  for (size_t i=0; i < pred_state.components; i++)
124  {
125  /* Evaluate the Kalman Gain
126  K = Pxy * (Py)^{-1} */
127  MatrixXd K = Pxy.middleCols(meas_size * i, meas_size) * pred_meas.covariance(i).inverse();
128 
129  /* Evaluate the filtered mean.

131  corr_state.mean(i) = pred_state.mean(i) + K * innovations.col(i);
132 
133  /* Evaluate the filtered covariance
134  P_{k}+ = P_{k}- - K * Py * K' */
135  corr_state.covariance(i) = pred_state.covariance(i) - K * pred_meas.covariance(i) * K.transpose();
136  }
